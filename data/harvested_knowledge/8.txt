@'
import hashlib
import json
import os
from datetime import datetime
from pathlib import Path

# IPFS integration
class IPFSProvenance:
    def __init__(self):
        self.gallery_path = Path(os.environ['USERPROFILE']) / 'Desktop' / 'GAZA_ROSE_EMPIRE' / 'gallery'
        self.nft_path = Path(os.environ['USERPROFILE']) / 'Desktop' / 'GAZA_ROSE_EMPIRE' / 'nft'
        self.nft_path.mkdir(exist_ok=True)
    
    def create_artwork_provenance(self, image_path, title, artist="Meeko"):
        """Generate immutable proof of creation with IPFS + SHA256"""
        with open(image_path, 'rb') as f:
            file_hash = hashlib.sha256(f.read()).hexdigest()
        
        provenance = {
            "title": title,
            "artist": artist,
            "created": datetime.now().isoformat(),
            "sha256": file_hash,
            "edition": 1,
            "collection": "Gaza Rose",
            "license": "CC BY-NC 4.0",
            "humanitarian": "100% of proceeds to UNRWA USA - Gaza Relief"
        }
        
        # Save provenance locally
        nft_file = self.nft_path / f"{title.replace(' ', '_')}.json"
        with open(nft_file, 'w') as f:
            json.dump(provenance, f, indent=2)
        
        # IPFS add (requires daemon)
        import subprocess
        result = subprocess.run(['ipfs', 'add', '-Q', str(image_path)], 
                              capture_output=True, text=True)
        ipfs_hash = result.stdout.strip()
        
        provenance['ipfs'] = f"ipfs://{ipfs_hash}"
        
        print(f"  ✅ Provenance created: {title}")
        print(f"     IPFS: {provenance['ipfs']}")
        print(f"     SHA256: {file_hash[:16]}...")
        
        return provenance

async function connectWhatsApp() {
    const { state, saveCreds } = await useMultiFileAuthState('auth_info');
    const sock = makeWASocket({
        printQRInTerminal: true,
        auth: state
    });

    sock.ev.on('creds.update', saveCreds);
    
    sock.ev.on('messages.upsert', async ({ messages }) => {
        const m = messages[0];
        if (!m.message) return;
        
        const text = m.message.conversation || m.message.extendedTextMessage?.text;
        
        if (text?.toLowerCase().includes('gallery')) {
            await sock.sendMessage(m.key.remoteJid, {
                text: '🖼️ GAZA ROSE GALLERY\nhttp://localhost:8000\nArtist: Meeko'
            });
        }
    });
