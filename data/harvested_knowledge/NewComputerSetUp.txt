B. Quick Fix Script (quick_fix.py):

#!/usr/bin/env python3
"""
SolarPunk Quick Fix
Save as: C:\Users\%USERNAME%\SolarPunk\connected\SolarPunk-Autonomous\quick_fix.py
"""

import os
import subprocess

def fix_all_repos():
    repos = [
        ("SolarPunk-Autonomous", r"C:\Users\%USERNAME%\SolarPunk\connected\SolarPunk-Autonomous"),
        ("SolarPunk-Nexus", r"C:\Users\%USERNAME%\SolarPunk\SolarPunk-Nexus"),
        ("SolarPunk-Memvid", r"C:\Users\%USERNAME%\SolarPunk\memvid_repo"),
        ("SolarPunk_Agent", r"C:\Users\%USERNAME%\SolarPunk\SolarPunk_Agent"),
        ("SolarPunk_Alpha_Bot", r"C:\Users\%USERNAME%\SolarPunk\SolarPunk_Alpha_Bot"),
        ("Murmuration", r"C:\Users\%USERNAME%\SolarPunk\Murmuration"),
        ("solarpunk-revenue", r"C:\Users\%USERNAME%\SolarPunk\solarpunk-revenue")
    ]
    
    for repo_name, repo_path in repos:
        if os.path.exists(repo_path):
            print(f"Fixing {repo_name}...")
            os.chdir(repo_path)
            subprocess.run(['git', 'remote', 'set-url', 'origin', 
                          f'https://github.com/MeekoThaRaccoon/{repo_name}.git'])
            print(f"  ✅ Set to MeekoThaRaccoon/{repo_name}")

if __name__ == "__main__":
    fix_all_repos()
