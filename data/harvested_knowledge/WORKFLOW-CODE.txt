name: "AI Auto-Fix Everything"
on:
  push:
    branches: [ master ]
  pull_request:
    branches: [ master ]
  schedule:
    - cron: '0 0 * * *'  # Daily at midnight
  workflow_dispatch:  # Manual trigger button

jobs:
  auto-fix:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4
      
      - name: Setup Node.js (if needed)
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          
      - name: AI Code Analysis and Fix
        uses: 1set/gpt-pull-request-review@v2.0.4
        with:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }}
          mode: "fix"
          
      - name: Create Fix Pull Request
        uses: peter-evans/create-pull-request@v5
        with:
          title: "🤖 AI Auto-Fix: Code improvements and fixes"
          body: |
            ## AI Auto-Fix Results
            - ✅ Code analysis completed
            - ✅ Suggested fixes applied
            - ✅ Automated by SolarPunk AI System
            - ✅ Triggered by workflow_dispatch
          branch: "ai-auto-fix-$(date +%s)"
          delete-branch: true
