## The Script

Save as `morning_briefing.py`:

```python
import smtplib, requests
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from datetime import datetime

GMAIL_ADDRESS = "your@gmail.com"
GMAIL_APP_PASSWORD = "your-app-password"
WEATHER_CITY = "New York"
NEWS_TOPICS = ["AI", "open source", "Gaza"]

def get_weather():
    r = requests.get(f"https://wttr.in/{WEATHER_CITY}?format=3",
        headers={"User-Agent": "Mozilla/5.0"})
    return r.text.strip()

def get_news():
    results = []
    for topic in NEWS_TOPICS:
        r = requests.get(f"https://news.google.com/rss/search?q={topic}&hl=en", timeout=5)
        import xml.etree.ElementTree as ET
        root = ET.fromstring(r.content)
        for item in root.findall('.//item')[:2]:
            results.append(f"  - {item.find('title').text}")
    return "\n".join(results[:6])

def get_ai_thought():
    try:
        r = requests.post("http://localhost:11434/api/generate", json={
            "model": "mistral",
            "prompt": f"One specific actionable idea for {datetime.now().strftime('%A %B %d')}. Under 50 words.",
            "stream": False
        }, timeout=30)
        return r.json()["response"]
    except:
        return "Start Ollama with: ollama serve"

def send_briefing():
    now = datetime.now()
    body = f"""MORNING BRIEFING — {now.strftime('%A, %B %d, %Y')}

WEATHER: {get_weather()}

NEWS:
{get_news()}

AI THOUGHT:
{get_ai_thought()}

Day {now.timetuple().tm_yday} of {now.year} — {365 - now.timetuple().tm_yday} days left.

Generated by Meeko Mycelium · github.com/meekotharaccoon-cell"""
    
    msg = MIMEMultipart()
    msg["From"] = GMAIL_ADDRESS
    msg["To"] = GMAIL_ADDRESS
    msg["Subject"] = f"Morning Briefing — {now.strftime('%b %d')}"
    msg.attach(MIMEText(body, "plain"))
    
    with smtplib.SMTP_SSL("smtp.gmail.com", 465) as s:
        s.login(GMAIL_ADDRESS, GMAIL_APP_PASSWORD)
        s.send_message(msg)
    print(f"Sent at {now.strftime('%H:%M')}")

if __name__ == "__main__":
    send_briefing()
```

## Schedule It

**Mac/Linux cron:**
```bash
0 8 * * * python3 /path/to/morning_briefing.py
```

**Windows Task Scheduler:** Daily at 8:00 AM → run `python morning_briefing.py`

**GitHub Actions (runs when computer is off):**
```yaml
name: Morning Briefing
on:
  schedule:
    - cron: '0 8 * * *'
jobs:
  brief:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: pip install requests && python morning_briefing.py
        env:
          GMAIL_ADDRESS: ${{ secrets.GMAIL_ADDRESS }}
          GMAIL_APP_PASSWORD: ${{ secrets.GMAIL_APP_PASSWORD }}
```
